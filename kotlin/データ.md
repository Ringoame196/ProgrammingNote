        val ymlFile = acquisitionYml(file)
        return ymlFile.getString(key)

        val ymlFile = acquisitionYml(file)

import java.io.File

class YamlFileManager {
    fun loadYAsMap(filePath: String): Map<String, String> {
        val yaml = Yaml()
        val inputStream = File(filePath).inputStream()
        return yaml.load(inputStream) as Map<String, String>
    }
}

/*

import org.bukkit.Bukkit
import java.sql.Connection
import java.sql.DriverManager
import java.sql.PreparedStatement
import java.sql.ResultSet
import java.sql.SQLException

object DataBaseManager {
    lateinit var dbFilePath: String

    // SQLiteコネクションを取得
    private val connection: Connection
        get() = DriverManager.getConnection("jdbc:sqlite:$dbFilePath")

    /**
     * SQLコマンドを実行する（INSERT, UPDATE, DELETE 用）
     */
    fun executeUpdate(command: String, parameters: List<Any>? = null) {
        try {
            connection.use { conn ->
                conn.prepareStatement(command).use { stmt ->
                    parameters?.bindParameters(stmt)
                    stmt.executeUpdate()
                }
            }
        } catch (e: SQLException) {
            Bukkit.getLogger().severe("SQL Error in executeUpdate: ${e.message}")
        }
    }

    /**
     * 単一の値を取得する
     */
    fun acquisitionValue(sql: String, parameters: List<Any> = emptyList(), label: String): Any? {
        val result = acquisitionValues(sql, parameters, listOf(label))
        return result[label]
    }

    /**
     * 単一行の複数カラム取得
     */
    fun acquisitionValues(sql: String, parameters: List<Any> = emptyList(), keys: List<String>): Map<String, Any?> {
        val values = mutableMapOf<String, Any?>()
        try {
            connection.use { conn ->
                conn.prepareStatement(sql).use { stmt ->
                    parameters.bindParameters(stmt)
                    stmt.executeQuery().use { rs ->
                        if (rs.next()) {
                            keys.forEach { key ->
                                values[key] = safeGetString(rs, key)
                            }
                        }
                    }
                }
            }
        } catch (e: SQLException) {
            Bukkit.getLogger().severe("SQL Error in acquisitionValues: ${e.message}")
        }
        return values
    }

    /**
     * 複数行取得
     */
    fun acquisitionValuesList(sql: String, parameters: List<Any> = emptyList(), keys: List<String>): List<Map<String, Any?>> {
        val results = mutableListOf<Map<String, Any?>>()
        try {
            connection.use { conn ->
                conn.prepareStatement(sql).use { stmt ->
                    parameters.bindParameters(stmt)
                    stmt.executeQuery().use { rs ->
                        while (rs.next()) {
                            val row = keys.associateWith { safeGetString(rs, it) }
                            results.add(row)
                        }
                    }
                }
            }
        } catch (e: SQLException) {
            Bukkit.getLogger().severe("SQL Error in acquisitionValuesList: ${e.message}")
        }
        return results
    }

    // PreparedStatement にパラメータを安全にバインド
    private fun List<Any>.bindParameters(stmt: PreparedStatement) {
        this.forEachIndexed { index, param ->
            stmt.setObject(index + 1, param)
        }
    }

    // ResultSet から安全に文字列取得（カラム存在しない場合 null）
    private fun safeGetString(rs: ResultSet, column: String): String? {
        return try {
            rs.getString(column)
        } catch (e: SQLException) {
            null
        }
    }
}
